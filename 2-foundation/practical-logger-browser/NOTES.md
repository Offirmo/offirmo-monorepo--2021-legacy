import { createLogger } from '@offirmo/practical-logger-browser'

const logger = createLogger({
	name: 'FOO',
	suggestedLevel: 'silly', // optional
})

logger.log('hello')

;[
	'fatal',
	'emerg',
	'alert',
	'crit',
	'error',
	'warning',
	'warn',
	'notice',
	'info',
	'verbose',
	'log',
	'debug',
	'trace',
	'silly',
].forEach(level => logger[level]({level}))
